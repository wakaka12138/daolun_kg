    # File lib/reline/general_io.rb, line 43
def self.getc(_timeout_second)
  unless @@buf.empty?
    return @@buf.shift
  end
  c = nil
  loop do
    Reline.core.line_editor.handle_signal
    result = @@input.wait_readable(0.1)
    next if result.nil?
    c = @@input.read(1)
    break
  end
  c&.ord
end

def self.reset(encoding: nil)

  elsif defined?(@@encoding)

  end
